/*
* DrawPixel draws a single pixel to the screen at the point in (px,py).
*/
void DrawPixel(unsigned int px,unsigned int py){

        register long int address = GRAPHIC_ADDRESS;

        /* Obtaining the height of frame buffer from the frame buffer structure */
        int height = *((unsigned int*)(address+4));
        height = height - 1 ;

        /* Validating the input*/
        if( py > height ){
                return;
        }

        /* Obtaining the width of frame buffer from the frame buffer structure */
        int width = *((unsigned int*)(address+0));
        width = width - 1;

        /* Validating the input*/
        if( px > width ){
                return;
        }

        /* Obtaining a pointer to frame buffer*/
        address = *((unsigned int*)(address+32));
        width = width + 1;
        px = (py * width) + px;
        address = address + (px << 1);
        *((unsigned short int*)(address)) = *((unsigned short int*)(FORE_COLOR));;
}

/*

* We wish to draw a line from (x0,y0) to (x1,y1), using only a function
* setPixel(x,y) which draws a dot in the pixel given by (x,y)
* Uses Bresenham's Line Algortihm which approximates a line
*/
void DrawLine(unsigned int x0,unsigned int y0,unsigned int x1,unsigned int y1){

        int dx=0, sx=0, dyn=0, sy=0, err=0;

         if( x0 > x1 ){
                 dx = x0 - x1;
                 sx = -1;
         }else{
                 dx = x1 - x0;
                 sx = 1;
         }

        if( y0 > y1 ){
                 dyn = y1 - y0;
                 sy = -1;
         }else{
                 dyn = y0 - y1;
                 sy = 1;
        }

        err = dx + dyn;
         x1 = x1 + sx;
         y1 = y1 + sy;

        while( 1 ){
                 if( x0 == x1 || y0 == y1 ){
                         return;
                 }

                /* Drawing pixel at location (x0,y0)*/
                 DrawPixel( x0, y0 );
                 if( dyn <= (err << 1) ){
                         err = err + dyn;
                         x0 = x0 + sx;
                 }
                 if( dx >= (err << 1) ){
                         err = err + dx;
                         y0 = y0 + sy;
                 }
         }
}

/*
* It draws a character starting at point (x,y). The character is displayed on the screen
* by actually displaying the monochrome image for that character stored in the bit map
* font provided above
*/
int *DrawCharacter(char character,unsigned int x,unsigned int y){
         static int result[2];

        /* Validating the input*/
         if(character > 127){
                  result[0] = result[1] = 0;
                  return result;
         }

        /* Calculatin the address of the character given as input in the bit map font given at the top*/
         unsigned char *charAddr = font_bin + (character << 4);
         int row;
         for(row = 0; row < 16; ++row){
                  unsigned char bits = *(charAddr + row);
                  int bit;
                  for(bit = 0; bit < 7; ++bit){
                           if( ((bits >> bit) & 0x1) ){

                                 /* Drawing pixel at location (x+bit,y+row)*/
                                   DrawPixel(x+bit,y+row);
                           }
                  }
         }

        /* Returning the width and height of the drawn character*/
         result[0] = 8;
         result[1] = 16;
         return result;
}
